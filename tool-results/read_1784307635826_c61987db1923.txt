    48тЖТ    48тЖТ * Vault payload stored in the GitHub repo.
    49тЖТ    49тЖТ *
    50тЖТ    50тЖТ * SECURITY MODEL
    51тЖТ    51тЖТ * --------------
    52тЖТ    52тЖТ * The plaintext `message` is NEVER stored in the repo. Only `ciphertext`
    53тЖТ    53тЖТ * (base64-encoded AES-GCM output) is persisted. The encryption key is
    54тЖТ    54тЖТ * derived from the owner password (OWNER_PASSWORD) using PBKDF2 with a
    55тЖТ    55тЖТ * per-vault random `salt` (16 bytes). A fresh random `iv` (12 bytes) is
    56тЖТ    56тЖТ * generated on every save.
    57тЖТ    57тЖТ *
    58тЖТ    58тЖТ * To decrypt, you need BOTH:
    59тЖТ    59тЖТ *   1. The owner password (only the owner knows it тАФ required to enter the
    60тЖТ    60тЖТ *      panel in the first place)
    61тЖТ    61тЖТ *   2. This website's JavaScript code (contains the algorithm parameters
    62тЖТ    62тЖТ *      below: PBKDF2 iterations, AES-GCM, key length)
    63тЖТ    63тЖТ *
    64тЖТ    64тЖТ * Even if someone reads the raw vault.json in GitHub, they only see
    65тЖТ    65тЖТ * opaque base64 ciphertext + the salt + iv. Without the password, the
    66тЖТ    66тЖТ * ciphertext cannot be decrypted in any app тАФ only this website can do it
    67тЖТ    67тЖТ * (because the password is required and is never transmitted or persisted).
    68тЖТ    68тЖТ *
    69тЖТ    69тЖТ * The `read` flag and `updatedAt` timestamp are stored in plaintext since
    70тЖТ    70тЖТ * they are not sensitive (just UI state).
    71тЖТ    71тЖТ */

    95тЖТ    95тЖТfunction bufToBase64(buf: ArrayBuffer | Uint8Array): string {
    96тЖТ    96тЖТ  const bytes = buf instanceof Uint8Array ? buf : new Uint8Array(buf);
    97тЖТ    97тЖТ  let binary = "";
    98тЖТ    98тЖТ  for (let i = 0; i < bytes.byteLength; i++) {
    99тЖТ    99тЖТ    binary += String.fromCharCode(bytes[i]);
   100тЖТ   100тЖТ  }
   101тЖТ   101тЖТ  return btoa(binary);
   102тЖТ   102тЖТ}
   103тЖТ   103тЖТ
   104тЖТ   104тЖТfunction base64ToBuf(b64: string): Uint8Array {
   105тЖТ   105тЖТ  const binary = atob(b64);
   106тЖТ   106тЖТ  const bytes = new Uint8Array(binary.length);
   107тЖТ   107тЖТ  for (let i = 0; i < binary.length; i++) {
   108тЖТ   108тЖТ    bytes[i] = binary.charCodeAt(i);
   109тЖТ   109тЖТ  }
   110тЖТ   110тЖТ  return bytes;
   111тЖТ   111тЖТ}

   113тЖТ   113тЖТasync function deriveKey(password: string, salt: Uint8Array): Promise<CryptoKey> {
   114тЖТ   114тЖТ  const enc = new TextEncoder();
   115тЖТ   115тЖТ  const baseKey = await crypto.subtle.importKey(
   116тЖТ   116тЖТ    "raw",
   117тЖТ   117тЖТ    enc.encode(password),
   118тЖТ   118тЖТ    { name: "PBKDF2" },
   119тЖТ   119тЖТ    false,
   120тЖТ   120тЖТ    ["deriveKey"]
   121тЖТ   121тЖТ  );
   122тЖТ   122тЖТ  return crypto.subtle.deriveKey(
   123тЖТ   123тЖТ    {
   124тЖТ   124тЖТ      name: "PBKDF2",
   125тЖТ   125тЖТ      salt: salt as unknown as BufferSource,
   126тЖТ   126тЖТ      iterations: PBKDF2_ITERATIONS,
   127тЖТ   127тЖТ      hash: "SHA-256",
   128тЖТ   128тЖТ    },
   129тЖТ   129тЖТ    baseKey,
   130тЖТ   130тЖТ    { name: "AES-GCM", length: KEY_LENGTH_BITS },
   131тЖТ   131тЖТ    false,
   132тЖТ   132тЖТ    ["encrypt", "decrypt"]
   133тЖТ   133тЖТ  );
   134тЖТ   134тЖТ}

   136тЖТ   136тЖТasync function encryptMessage(plaintext: string, password: string): Promise<{
   137тЖТ   137тЖТ  salt: string;
   138тЖТ   138тЖТ  iv: string;
   139тЖТ   139тЖТ  ciphertext: string;
   140тЖТ   140тЖТ}> {
   141тЖТ   141тЖТ  // Generate fresh salt + iv for this save
   142тЖТ   142тЖТ  const salt = crypto.getRandomValues(new Uint8Array(SALT_BYTES));
   143тЖТ   143тЖТ  const iv = crypto.getRandomValues(new Uint8Array(IV_BYTES));
   144тЖТ   144тЖТ  const key = await deriveKey(password, salt);
   145тЖТ   145тЖТ  const enc = new TextEncoder();
   146тЖТ   146тЖТ  const ctBuf = await crypto.subtle.encrypt(
   147тЖТ   147тЖТ    { name: "AES-GCM", iv: iv as unknown as BufferSource },
   148тЖТ   148тЖТ    key,
   149тЖТ   149тЖТ    enc.encode(plaintext)
   150тЖТ   150тЖТ  );
   151тЖТ   151тЖТ  return {
   152тЖТ   152тЖТ    salt: bufToBase64(salt),
   153тЖТ   153тЖТ    iv: bufToBase64(iv),
   154тЖТ   154тЖТ    ciphertext: bufToBase64(ctBuf),
   155тЖТ   155тЖТ  };
   156тЖТ   156тЖТ}

   158тЖТ   158тЖТasync function decryptMessage(
   159тЖТ   159тЖТ  ciphertextB64: string,
   160тЖТ   160тЖТ  ivB64: string,
   161тЖТ   161тЖТ  saltB64: string,
   162тЖТ   162тЖТ  password: string
   163тЖТ   163тЖТ): Promise<string> {
   164тЖТ   164тЖТ  const salt = base64ToBuf(saltB64);
   165тЖТ   165тЖТ  const iv = base64ToBuf(ivB64);
   166тЖТ   166тЖТ  const key = await deriveKey(password, salt);
   167тЖТ   167тЖТ  const ct = base64ToBuf(ciphertextB64);
   168тЖТ   168тЖТ  const ptBuf = await crypto.subtle.decrypt(
   169тЖТ   169тЖТ    { name: "AES-GCM", iv: iv as unknown as BufferSource },
   170тЖТ   170тЖТ    key,
   171тЖТ   171тЖТ    ct as unknown as BufferSource
   172тЖТ   172тЖТ  );
   173тЖТ   173тЖТ  const dec = new TextDecoder();
   174тЖТ   174тЖТ  return dec.decode(ptBuf);
   175тЖТ   175тЖТ}

   332тЖТ   332тЖТ      let decryptedMessage = "";
   333тЖТ   333тЖТ      let parsedRead = false;

   335тЖТ   335тЖТ

   344тЖТ   344тЖТ          );

   351тЖТ   351тЖТ          decryptedMessage = "";
   352тЖТ   352тЖТ          parsedRead = !!parsed.read;
   353тЖТ   353тЖТ          parsedTs = parsed.updatedAt || "";
   354тЖТ   354тЖТ        }

   383тЖТ   383тЖТ  async function saveToCloud() {
   384тЖТ   384тЖТ    setSyncState("saving");
   385тЖТ   385тЖТ
   386тЖТ   386тЖТ    // Encrypt the message using the active password before storing.
   387тЖТ   387тЖТ    // Even if the repo is leaked, the message cannot be read without the password.
   388тЖТ   388тЖТ    let salt: string, iv: string, ciphertext: string;
   389тЖТ   389тЖТ    try {
   390тЖТ   390тЖТ      const enc = await encryptMessage(message, activePassword);
   391тЖТ   391тЖТ      salt = enc.salt;
   392тЖТ   392тЖТ      iv = enc.iv;

   396тЖТ   396тЖТ      setSyncState("error");

   398тЖТ   398тЖТ      return;

   401тЖТ   401тЖТ    const payload: VaultPayload = {

   404тЖТ   404тЖТ      iv,
   405тЖТ   405тЖТ      ciphertext,

   410тЖТ   410тЖТ

   412тЖТ   412тЖТ    // locally so the UI can show it instantly on next unlock

   414тЖТ   414тЖТ      localStorage.setItem(CACHE_MSG, message);
   415тЖТ   415тЖТ      localStorage.setItem(CACHE_READ, readChecked ? "true" : "false");

   418тЖТ   418тЖТ      /* ignore */

   458тЖТ   458тЖТ      }
